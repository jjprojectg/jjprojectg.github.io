      <img src="http://www.foodsafetykorea.go.kr/uploadimg/cook/853-2.jpg" alt="단계 1 이미지" class="step-image"/>

      <img src="http://www.foodsafetykorea.go.kr/uploadimg/cook/853-3.jpg" alt="단계 1 이미지" class="step-image"/>
